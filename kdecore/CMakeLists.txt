    SOURCES
        ${libkdecore_HEADERS}
        ${libkdecore_NOINST_HEADERS}
    )

    SOURCES
        kasyncio.h
        kdesktopwidget.h
        )

    SOURCES
        ${kdecore_SRCS}
    PRIVATE_LIBS
        ltdl::ltdl
    COMPILE_OPTIONS
        -Wno-format-security
    LIBS
        Qt::Qt2
        kde2::DCOP
        ${X11_LIBRARIES}
    VERSION
        3.0.0
    )
